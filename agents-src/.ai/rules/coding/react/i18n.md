name: i18n
Scope: When adding user-facing strings, translating messages, or configuring internationalization
description: All user-facing strings go through react-i18next; never hardcode visible text
---

Use `react-i18next` for all internationalization. Never hardcode user-facing strings — all visible text must go through the i18n layer.

## Setup

```bash
npm install i18next react-i18next i18next-http-backend i18next-browser-languagedetector
npm install -D @types/i18next
```

Configure i18next in `src/lib/i18n.ts`:

```ts
// src/lib/i18n.ts
import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import HttpBackend from 'i18next-http-backend'
import LanguageDetector from 'i18next-browser-languagedetector'

i18n
  .use(HttpBackend)
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    fallbackLng: 'pt-BR',
    supportedLngs: ['pt-BR', 'en'],
    backend: { loadPath: '/locale/{{lng}}/{{ns}}.json' },
    interpolation: { escapeValue: false },
  })

export default i18n
```

Import once in `main.tsx` before rendering:

```ts
import '@/lib/i18n'
```

## TypeScript key safety

Define the resource type so TypeScript validates all translation keys:

```ts
// src/lib/i18n.d.ts
import ptBR from '../../public/locale/pt-BR/translation.json'

declare module 'i18next' {
  interface CustomTypeOptions {
    defaultNS: 'translation'
    resources: { translation: typeof ptBR }
  }
}
```

With this, `t('nonexistent.key')` is a TypeScript error.

## Usage in components

```tsx
import { useTranslation } from 'react-i18next'

export function SaveButton() {
  const { t } = useTranslation()
  return <Button>{t('common.save')}</Button>
}
```

## Locale files

Translation files live in `public/locale/` so they are served statically and loaded on demand:

```
public/locale/
  pt-BR/
    translation.json
  en/
    translation.json
```

Organize keys by page and component — never by domain/DDD concept. Each page and component owns its keys under its own namespace:

```json
{
  "common": { "save": "Salvar", "cancel": "Cancelar", "confirm": "Confirmar" },
  "error": {
    "validation": "Dados inválidos",
    "forbidden": "Sem permissão para acessar este recurso",
    "server": "Erro interno do servidor"
  },
  "page": {
    "produtoList": { "title": "Produtos", "empty": "Nenhum produto encontrado" },
    "produtoForm": { "title": "Novo Produto", "submit": "Salvar produto" }
  },
  "component": {
    "header": { "logout": "Sair", "profile": "Perfil" },
    "productCard": { "addToCart": "Adicionar ao carrinho" }
  }
}
```

Never group keys by domain (`produto.*`, `pedido.*`) — that couples the translation structure to the data model instead of the UI structure.

## Error messages

`useQueryError` uses `t()` from `useTranslation` to return translated error messages:

```ts
// src/hook/error/useQueryError.ts
import { useEffect } from 'react'
import { useTranslation } from 'react-i18next'
import { useErrorStore } from '@/store/error/error.store'
import { ValidationError, ForbiddenError, ServerError } from '@/service/api/main.httpClient'

export function useQueryError(error: Error | null) {
  const addError = useErrorStore(s => s.addError)
  const { t } = useTranslation()

  useEffect(() => {
    if (!error) return
    if (error instanceof ValidationError) addError(t('error.validation'))
    else if (error instanceof ForbiddenError) addError(t('error.forbidden'))
    else if (error instanceof ServerError) addError(t('error.server'))
  }, [error, addError, t])
}
```

No build step required — edit `.json` files directly and the changes are picked up on the next request.
